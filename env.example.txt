# Supabase 配置
NEXT_PUBLIC_SUPABASE_URL=your-supabase-project-url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-supabase-anon-key

# QQ邮箱 SMTP 配置
SMTP_HOST=smtp.qq.com
SMTP_PORT=587
SMTP_USER=your_qq_number@qq.com
SMTP_PASS=your-16-digit-authorization-code

# 将此文件重命名为 .env.local 并填入真实配置
# 注意：
# 1. SMTP_PASS 是QQ邮箱的16位授权码，不是登录密码
# 2. 获取授权码：QQ邮箱 → 设置 → 账户 → 开启SMTP服务 → 生成授权码
# 3. SMTP_USER 必须与发件人邮箱一致
